<objective>
Create the auth state machine — the heart of Phase 3. Split into:

1. A **pure reducer** `reduce(state, event, ctx) → {next, sideEffects[]}` with NO electron
   import, NO filesystem, NO timers. Testable with plain object inputs.
2. A **thin executor** `notify(event)` that maintains module-scoped `currentState`,
   `attempts`, `creds`, `webContents`, `store`, `safeStorage`, `mainWindow`, `timers`, and
   processes the side-effect array.
3. A **start(opts)** bootstrap that wires the executor, synchronously runs the
   safeStorage-check + credentials-load sequence, and kicks the reducer with a `boot` event.
4. A **`handleCredentialsSubmit(payload)` IPC handler** that persists both the admin PIN
   hash record AND the credentials ciphertext in a **single atomic `store.set({...})` call**,
   composed from `adminPin.buildRecord(pin)` and `credentialsStore.buildCiphertext(safeStorage, creds)`.
   See `<read_first>` on Task 1 for the exact code shape — the two-call anti-pattern
   (`adminPin.setPin` then `credentialsStore.saveCredentials`) is explicitly forbidden
   because a crash between calls leaves `adminPin` set while `credentialsCiphertext` is
   missing, which the boot path mis-detects as "first run", silently over-writing the
   stale PIN hash (unrecoverable without manual RDP edit of config.json).

Purpose: AUTH-03 (auto-fill + submit), AUTH-04 (reactive state machine with logged
transitions), AUTH-05 (decrypt-failed / safestorage-unavailable recovery path via PIN gate).

Output: `src/main/authFlow.js`, `test/authFlow.test.js`.

    6. **pin-attempt handling:** The reducer does NOT handle raw PIN strings (would leak into
       logs if someone stringified ctx). Instead the executor has a `notifyPinAttempt(pin)`
       wrapper that calls adminPin.verifyPin → converts to `{type:'pin-ok'}` or `{type:'pin-bad'}`
       → calls reduce. The raw pin never reaches the reducer.

       overlay): executor handles by sending `show-pin-modal` IPC to mainWindow. Not a reducer event.
    8. **`rerun-boot` side effect** (from NEEDS_CREDENTIALS + credentials-submitted): executor
       re-loads credentials from the store (caller just saved them), re-sets module-scoped
       `creds`, resets `attempts = 0`, clears `authFailedActive`, sends `hide-credentials-overlay`
       IPC, then dispatches `{type:'creds-loaded'}` — Magicline should still be sitting at the
       login page, so a fresh `login-detected` will fire from the still-running drain.
    9. **Log format:** `log.info('auth.state: ' + prev + ' -> ' + next + ' reason=' + reason)`
       per D-18. NEVER log creds, pin, or ciphertext length. The `log` side effect in the
       reducer just emits a reason — the state-transition log line is emitted by `notify` itself
       when `next !== currentState`.

    // src/main/authFlow.js
    // -----------------------------------------------------------------------------
    // Phase 3 auth state machine. Pure reducer + thin executor.
    //
    // The reducer (reduce) has NO side effects. It is exported for unit testing
    // without Electron. The executor holds module-scoped state (currentState,
    // attempts, creds, wc, timers) and is the ONLY place that talks to
    // webContents, credentialsStore, adminPin, and setTimeout.
    //
    // States: BOOTING → NEEDS_CREDENTIALS → LOGIN_DETECTED → LOGIN_SUBMITTED
    //       → CASH_REGISTER_READY | LOGIN_FAILED | CREDENTIALS_UNAVAILABLE
    // -----------------------------------------------------------------------------

    const log = require('./logger');
    const credentialsStore = require('./credentialsStore');
    const adminPin = require('./adminPin');

    const STATES = Object.freeze({
      BOOTING:                 'BOOTING',
      NEEDS_CREDENTIALS:       'NEEDS_CREDENTIALS',
      LOGIN_DETECTED:          'LOGIN_DETECTED',
      LOGIN_SUBMITTED:         'LOGIN_SUBMITTED',
      CASH_REGISTER_READY:     'CASH_REGISTER_READY',
      LOGIN_FAILED:            'LOGIN_FAILED',
      CREDENTIALS_UNAVAILABLE: 'CREDENTIALS_UNAVAILABLE',
    });

    const MAX_ATTEMPTS         = 3;
    const POST_SUBMIT_WATCHDOG_MS = 8000;
    const BOOT_WATCHDOG_MS        = 12000;
    const RETRY_BACKOFF_MS        = 2000;

    // -----------------------------------------------------------------------------
    // PURE REDUCER
    // -----------------------------------------------------------------------------

    function retryOrFail(ctx, reason) {
      if (ctx.attempts + 1 >= ctx.maxAttempts) {
        return {
          next: STATES.LOGIN_FAILED,
          sideEffects: [
            { kind: 'log', reason: 'max-retries' },
            { kind: 'show-error', variant: 'login-failed' },
            { kind: 'set-failed-active' },
          ],
        };
      }
      return {
        next: STATES.LOGIN_DETECTED,
        sideEffects: [
          { kind: 'log', reason: 'retry:' + reason },
          { kind: 'increment-attempts' },
          { kind: 'start-timer', name: 'retry-backoff', ms: RETRY_BACKOFF_MS },
        ],
      };
    }

    function reduce(state, event, ctx) {
      switch (state) {
        case STATES.BOOTING: {
          if (event.type === 'creds-loaded') {
            if (ctx.hasCreds) {
              return {
                next: STATES.BOOTING,
                sideEffects: [
                  { kind: 'log', reason: 'creds-loaded' },
                  { kind: 'start-timer', name: 'boot', ms: BOOT_WATCHDOG_MS },
                ],
              };
            }
            return {
              next: STATES.NEEDS_CREDENTIALS,
              sideEffects: [
                { kind: 'log', reason: 'first-run' },
                { kind: 'show-credentials-overlay', firstRun: true },
              ],
            };
          }
          if (event.type === 'safestorage-unavailable') {
            return {
              next: STATES.CREDENTIALS_UNAVAILABLE,
              sideEffects: [
                { kind: 'log', reason: 'safestorage-unavailable' },
                { kind: 'show-error', variant: 'credentials-unavailable' },
              ],
            };
          }
          if (event.type === 'decrypt-failed') {
            return {
              next: STATES.CREDENTIALS_UNAVAILABLE,
              sideEffects: [
                { kind: 'log', reason: 'decrypt-failed' },
                { kind: 'show-error', variant: 'credentials-unavailable' },
              ],
            };
          }
          if (event.type === 'login-detected' && ctx.hasCreds) {
            return {
              next: STATES.LOGIN_DETECTED,
              sideEffects: [
                { kind: 'log', reason: 'login-detected' },
                { kind: 'clear-timer', name: 'boot' },
                { kind: 'fill-and-submit' },
                { kind: 'start-timer', name: 'post-submit', ms: POST_SUBMIT_WATCHDOG_MS },
              ],
            };
          }
          if (event.type === 'cash-register-ready') {
            // Pitfall #10: cookie-session skip-login path — Magicline served the
            // cash register directly without a login page appearance.
            return {
              next: STATES.CASH_REGISTER_READY,
              sideEffects: [
                { kind: 'log', reason: 'cash-register-ready-cookie' },
                { kind: 'clear-timer', name: 'boot' },
                { kind: 'reset-attempts' },
              ],
            };
          }
          if (event.type === 'timer-expired' && event.name === 'boot') {
            return {
              next: STATES.LOGIN_FAILED,
              sideEffects: [
                { kind: 'log', reason: 'boot-watchdog-expired' },
                { kind: 'show-error', variant: 'login-failed' },
                { kind: 'set-failed-active' },
              ],
            };
          }
          return { next: state, sideEffects: [] };
        }

        case STATES.LOGIN_DETECTED: {
          if (event.type === 'login-submitted') {
            return {
              next: STATES.LOGIN_SUBMITTED,
              sideEffects: [{ kind: 'log', reason: 'submit-fired' }],
            };
          }
          if (event.type === 'timer-expired' && event.name === 'post-submit') {
            return retryOrFail(ctx, 'watchdog-expired');
          }
          if (event.type === 'timer-expired' && event.name === 'retry-backoff') {
            return {
              next: STATES.LOGIN_DETECTED,
              sideEffects: [
                { kind: 'log', reason: 'retry-fire' },
                { kind: 'fill-and-submit' },
                { kind: 'start-timer', name: 'post-submit', ms: POST_SUBMIT_WATCHDOG_MS },
              ],
            };
          }
          return { next: state, sideEffects: [] };
        }

        case STATES.LOGIN_SUBMITTED: {
          if (event.type === 'cash-register-ready') {
            return {
              next: STATES.CASH_REGISTER_READY,
              sideEffects: [
                { kind: 'log', reason: 'cash-register-ready' },
                { kind: 'clear-timer', name: 'post-submit' },
                { kind: 'reset-attempts' },
              ],
            };
          }
          if (event.type === 'login-detected') {
            return retryOrFail(ctx, 'login-redetected');
          }
          if (event.type === 'timer-expired' && event.name === 'post-submit') {
            return retryOrFail(ctx, 'watchdog-expired');
          }
          return { next: state, sideEffects: [] };
        }

        case STATES.NEEDS_CREDENTIALS: {
          if (event.type === 'credentials-submitted') {
            return {
              next: STATES.BOOTING,
              sideEffects: [
                { kind: 'log', reason: 'creds-saved' },
                { kind: 'hide-credentials-overlay' },
                { kind: 'rerun-boot' },
              ],
            };
          }
          return { next: state, sideEffects: [] };
        }

        case STATES.LOGIN_FAILED:
        case STATES.CREDENTIALS_UNAVAILABLE: {
          if (event.type === 'pin-recovery-requested') {
            return {
              next: state,
              sideEffects: [{ kind: 'show-pin-modal' }],
            };
          }
          if (event.type === 'pin-ok') {
            return {
              next: STATES.NEEDS_CREDENTIALS,
              sideEffects: [
                { kind: 'log', reason: 'pin-ok' },
                { kind: 'hide-pin-modal' },
                { kind: 'clear-failed-active' },
                { kind: 'show-credentials-overlay', firstRun: false },
              ],
            };
          }
          if (event.type === 'pin-bad') {
            return {
              next: state,
              sideEffects: [{ kind: 'log', reason: 'pin-bad' }],
            };
          }
          return { next: state, sideEffects: [] };
        }

        case STATES.CASH_REGISTER_READY:
        default:
          return { next: state, sideEffects: [] };
      }
    }

    // -----------------------------------------------------------------------------
    // EXECUTOR (module-scoped state + side-effect runner)
    // -----------------------------------------------------------------------------

    let currentState = STATES.BOOTING;
    let attempts = 0;
    let creds = null;
    let mainWindow = null;
    let wc = null;
    let store = null;
    let safeStorageRef = null;
    let timers = {};
    let authFailedActive = false;
    let started = false;

    function notify(event) {
      const ctx = { hasCreds: !!creds, attempts: attempts, maxAttempts: MAX_ATTEMPTS };
      const { next, sideEffects } = reduce(currentState, event, ctx);
      if (next !== currentState) {
        log.info('auth.state: ' + currentState + ' -> ' + next + ' reason=' + (event.type || '?'));
        currentState = next;
      }
      for (const sx of sideEffects) {
        try {
          runSideEffect(sx);
        } catch (e) {
          log.error('authFlow.runSideEffect failed: ' + (sx.kind) + ' — ' + (e && e.message));
        }
      }
    }

    function runSideEffect(sx) {
      switch (sx.kind) {
        case 'log':
          log.info('auth.reason=' + sx.reason);
          return;

        case 'show-error':
          if (mainWindow && !mainWindow.isDestroyed()) {
            mainWindow.webContents.send('show-magicline-error', { variant: sx.variant });
          }
          return;

        case 'show-credentials-overlay':
          if (mainWindow && !mainWindow.isDestroyed()) {
            mainWindow.webContents.send('show-credentials-overlay', { firstRun: !!sx.firstRun });
          }
          return;

        case 'hide-credentials-overlay':
          if (mainWindow && !mainWindow.isDestroyed()) {
            mainWindow.webContents.send('hide-credentials-overlay');
          }
          return;

        case 'show-pin-modal':
          if (mainWindow && !mainWindow.isDestroyed()) {
            mainWindow.webContents.send('show-pin-modal');
          }
          return;

        case 'hide-pin-modal':
          if (mainWindow && !mainWindow.isDestroyed()) {
            mainWindow.webContents.send('hide-pin-modal');
          }
          return;

        case 'fill-and-submit':
          if (!wc || wc.isDestroyed() || !creds) return;
          // CRITICAL: JSON.stringify both arguments — Pitfall #6.
          const expr = 'window.__bskiosk_fillAndSubmitLogin(' +
            JSON.stringify(creds.user) + ',' + JSON.stringify(creds.pass) + ')';
          wc.executeJavaScript(expr, true).catch((err) => {
            log.error('authFlow.fill-and-submit executeJavaScript failed: ' + (err && err.message));
          });
          return;

        case 'increment-attempts':
          attempts += 1;
          return;

        case 'reset-attempts':
          attempts = 0;
          authFailedActive = false;
          return;

        case 'set-failed-active':
          authFailedActive = true;
          return;

        case 'clear-failed-active':
          authFailedActive = false;
          return;

        case 'start-timer':
          if (timers[sx.name]) clearTimeout(timers[sx.name]);
          timers[sx.name] = setTimeout(() => {
            timers[sx.name] = null;
            notify({ type: 'timer-expired', name: sx.name });
          }, sx.ms);
          return;

        case 'clear-timer':
          if (timers[sx.name]) {
            clearTimeout(timers[sx.name]);
            timers[sx.name] = null;
          }
          return;

        case 'rerun-boot':
          // Credentials were just saved by the IPC handler. Re-read them, reset
          // counters, and dispatch a fresh creds-loaded event. Magicline is still
          // at the login page so the still-running drain will fire login-detected
          // again shortly.
          try {
            const loaded = credentialsStore.loadCredentials(store, safeStorageRef);
            if (loaded && loaded !== credentialsStore.DECRYPT_FAILED) {
              creds = loaded;
              attempts = 0;
              authFailedActive = false;
              notify({ type: 'creds-loaded' });
            } else {
              log.error('authFlow.rerun-boot: loadCredentials returned ' + loaded);
              notify({ type: 'decrypt-failed' });
            }
          } catch (e) {
            log.error('authFlow.rerun-boot threw: ' + (e && e.message));
            notify({ type: 'decrypt-failed' });
          }
          return;

        default:
          log.warn('authFlow.runSideEffect: unknown kind=' + sx.kind);
      }
    }

    // PUBLIC ENTRY POINTS
    // -----------------------------------------------------------------------------

    function start(opts) {
      if (started) {
        log.warn('authFlow.start: already started — ignoring');
        return;
      }
      if (!opts || !opts.mainWindow || !opts.magiclineWebContents || !opts.store || !opts.safeStorage) {
        throw new Error('authFlow.start: opts must be {mainWindow, magiclineWebContents, store, safeStorage}');
      }
      mainWindow = opts.mainWindow;
      wc = opts.magiclineWebContents;
      store = opts.store;
      safeStorageRef = opts.safeStorage;
      started = true;
      currentState = STATES.BOOTING;
      attempts = 0;
      authFailedActive = false;

      log.info('authFlow.start: entering BOOTING');

      // Step 1: check safeStorage availability
      if (!credentialsStore.isStoreAvailable(safeStorageRef)) {
        notify({ type: 'safestorage-unavailable' });
        return;
      }

      // Step 2: load credentials
      let loaded;
      try {
        loaded = credentialsStore.loadCredentials(store, safeStorageRef);
      } catch (e) {
        log.error('authFlow.start: loadCredentials threw: ' + (e && e.message));
        notify({ type: 'decrypt-failed' });
        return;
      }

      if (loaded === credentialsStore.DECRYPT_FAILED) {
        notify({ type: 'decrypt-failed' });
        return;
      }
      if (loaded === null) {
        // First run — no credentials stored
        creds = null;
        notify({ type: 'creds-loaded' });
        return;
      }
      // Successfully loaded
      creds = loaded;
      notify({ type: 'creds-loaded' });
    }

    // ---------------------------------------------------------------------------
    // ATOMIC multi-key persist handler — CONTEXT.md D-11 / RESEARCH §electron-store Atomicity.
    //
    // The two-call anti-pattern (adminPin.setPin then credentialsStore.saveCredentials)
    // is FORBIDDEN here. If the second call crashes, boot next time mis-detects
    // the stale `adminPin` as "first run", silently over-writes the PIN, and the
    // admin is locked out of recovery (unrecoverable without manual RDP edit).
    //
    // Instead: pre-compute BOTH records via the pure builders (which do not touch
    // the store), then persist them in a SINGLE whole-file-atomic store.set({...})
    // call. electron-store writes temp file + rename → either both keys land or
    // neither does.
    // ---------------------------------------------------------------------------
    function handleCredentialsSubmit(payload) {
      // Called by main.js IPC handler for 'submit-credentials'.
      // payload: {firstRun, pin?, user, pass}
      if (!payload || typeof payload.user !== 'string' || typeof payload.pass !== 'string') {
        log.error('authFlow.handleCredentialsSubmit: invalid payload');
        return { ok: false, error: 'invalid-payload' };
      }
      if (payload.firstRun && typeof payload.pin !== 'string') {
        return { ok: false, error: 'pin-required-first-run' };
      }

      // Phase A — pre-compute BOTH records WITHOUT touching the store.
      let adminPinRecord = null;
      let credentialsCiphertext = null;
      try {
        if (payload.firstRun) {
          adminPinRecord = adminPin.buildRecord(payload.pin);
        }
        credentialsCiphertext = credentialsStore.buildCiphertext(
          safeStorageRef,
          { user: payload.user, pass: payload.pass }
        );
      } catch (err) {
        log.error('auth.submit.build-failed: ' + (err && err.message));
        if (mainWindow && !mainWindow.isDestroyed()) {
          mainWindow.webContents.send('credentials-submit-error', { reason: 'build-failed' });
        }
        return { ok: false, error: 'build-failed' };
      }

      // Phase B — single atomic store.set with BOTH keys in ONE object.
      // electron-store writes this as a temp file + rename, so it is whole-file
      // atomic. Either both keys land or neither does. Partial persist impossible.
      try {
        const update = { credentialsCiphertext: credentialsCiphertext };
        if (payload.firstRun) {
          update.adminPin = adminPinRecord;
        }
        store.set(update);
      } catch (err) {
        log.error('auth.submit.persist-failed: ' + (err && err.message));
        if (mainWindow && !mainWindow.isDestroyed()) {
          mainWindow.webContents.send('credentials-submit-error', { reason: 'persist-failed' });
        }
        return { ok: false, error: 'persist-failed' };
      }

      log.info('auth.submit.persisted: firstRun=' + (!!payload.firstRun));
      // Transition back to BOOTING via the reducer — rerun-boot side effect
      // will re-read the store and resume the flow.
      notify({ type: 'credentials-submitted' });
      return { ok: true };
    }

    function handlePinAttempt(pin) {
      try {
        const ok = adminPin.verifyPin(store, pin);
        notify({ type: ok ? 'pin-ok' : 'pin-bad' });
        return { ok: ok };
      } catch (e) {
        log.error('authFlow.handlePinAttempt threw: ' + (e && e.message));
        notify({ type: 'pin-bad' });
        return { ok: false };
      }
    }

    function handlePinRecoveryRequested() {
      notify({ type: 'pin-recovery-requested' });
    }

    }
