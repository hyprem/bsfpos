    ```js
    // src/main/sessionReset.js
    const { session } = require('electron');
    const log = require('./logger');

    const RESET_WINDOW_MS = 60_000;   // IDLE-05 rolling window
    const RESET_LOOP_THRESHOLD = 3;   // IDLE-05: >3 within 60s trips guard

    let resetting = false;
    let loopActive = false;
    const resetTimestamps = [];  // Array<{ t: number, reason: string }>

    let mainWindow = null;
    let store = null;

    function init(opts) {
      mainWindow = opts.mainWindow;
      store = opts.store;
    }

    async function hardReset({ reason }) {
      // D-15 step 1: in-flight or loop-active guard
      if (resetting || loopActive) {
        log.info('sessionReset.suppressed: ' + (resetting ? 'in-flight' : 'loop-active') + ' reason=' + reason);
        return;
      }
      if (!mainWindow) {
        throw new Error('sessionReset.hardReset: init({mainWindow, store}) was never called');
      }

      // D-15 step 2: rolling-window loop detection (D-17/D-18)
      const now = Date.now();
      const recent = resetTimestamps.filter(e => now - e.t < RESET_WINDOW_MS);
      resetTimestamps.length = 0;
      resetTimestamps.push(...recent, { t: now, reason });
      if (recent.length + 1 >= RESET_LOOP_THRESHOLD) {
        loopActive = true;
        log.error('sessionReset.loop-detected: count=' + (recent.length + 1) +
                  ' reasons=' + JSON.stringify(resetTimestamps.map(x => x.reason)));
        mainWindow.webContents.send('show-magicline-error', { variant: 'reset-loop' });
        return;
      }

      log.info('sessionReset.hardReset: reason=' + reason + ' count=' + (recent.length + 1));

      // D-15 step 3: in-flight guard SET
      resetting = true;
      try {
        // Step 4
        require('./idleTimer').stop();
        // Step 5 — CRITICAL: channel is 'splash:show' NOT 'show-splash' (RESEARCH pitfall 3)
        mainWindow.webContents.send('splash:show');
        // Step 6
        const { destroyMagiclineView, createMagiclineView } = require('./magiclineView');
        destroyMagiclineView(mainWindow);
        // Step 7
        const sess = session.fromPartition('persist:magicline');
        // Step 8 — exactly 6 storages (D-15)
        await sess.clearStorageData({
          storages: [
            'cookies',
            'localstorage',
            'sessionstorage',
            'serviceworkers',
            'indexdb',
            'cachestorage'
          ]
        });
        // Step 9 — flush BEFORE createMagiclineView (Pattern 4, Assumption A5)
        await sess.cookies.flushStore();
        // Step 10
        createMagiclineView(mainWindow, store);
      } finally {
        // Step 11
        resetting = false;
      }
    }

    function _resetForTests() {
      resetting = false;
      loopActive = false;
      resetTimestamps.length = 0;
      mainWindow = null;
      store = null;
    }

    function _getStateForTests() {
      return { resetting, loopActive, resetTimestamps: resetTimestamps.slice() };
    }

    module.exports = {
      init,
      hardReset,
      _resetForTests,
      _getStateForTests,
      _RESET_WINDOW_MS: RESET_WINDOW_MS,
      _RESET_LOOP_THRESHOLD: RESET_LOOP_THRESHOLD
    };
    ```
