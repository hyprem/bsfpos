```js
// tests/main/sessionReset.welcome-harness.test.js
// Phase 6 IDLE-04 extension: 5-cycle welcome-mode hardReset harness.

const path = require('path');

describe('Phase 6: 5-cycle welcome-mode harness', () => {
  let sessionReset;
  let clearStorageSpy;
  let flushStoreSpy;
  let destroySpy;
  let createSpy;
  let sendSpy;
  let preSpy;
  let postSpy;

  beforeEach(() => {
    jest.resetModules();

    clearStorageSpy = jest.fn().mockResolvedValue();
    flushStoreSpy   = jest.fn().mockResolvedValue();
    sendSpy         = jest.fn();

    jest.doMock('electron', () => ({
      session: {
        fromPartition: () => ({
          clearStorageData: clearStorageSpy,
          cookies: {
            get: jest.fn().mockResolvedValue([]),
            set: jest.fn().mockResolvedValue(),
            flushStore: flushStoreSpy,
          },
        }),
      },
    }));

    destroySpy = jest.fn();
    createSpy  = jest.fn();
    jest.doMock(path.join('..', '..', 'src', 'main', 'magiclineView'), () => ({
      destroyMagiclineView: destroySpy,
      createMagiclineView:  createSpy,
    }));

    jest.doMock(path.join('..', '..', 'src', 'main', 'idleTimer'), () => ({
      stop: jest.fn(),
    }));

    jest.doMock(path.join('..', '..', 'src', 'main', 'logger'), () => ({
      info: jest.fn(), warn: jest.fn(), error: jest.fn(), audit: jest.fn(),
    }));

    sessionReset = require('../../src/main/sessionReset');
    sessionReset.init({
      mainWindow: { webContents: { send: sendSpy } },
      store: { get: jest.fn(), set: jest.fn() },
    });

    preSpy  = jest.fn();
    postSpy = jest.fn();
    sessionReset.onPreReset(preSpy);
    sessionReset.onPostReset(postSpy);
  });

  afterEach(() => {
    sessionReset._resetForTests();
  });

  it('survives 5 consecutive welcome-mode resets without tripping loop guard', async () => {
    for (let i = 0; i < 5; i++) {
      await sessionReset.hardReset({ reason: 'idle-expired', mode: 'welcome' });
    }

    // Assertion 1 — storage wipe count + shape
    expect(clearStorageSpy).toHaveBeenCalledTimes(5);
    for (const call of clearStorageSpy.mock.calls) {
      const storages = call[0].storages;
      expect(storages).toEqual(expect.arrayContaining([
        'cookies', 'localstorage', 'sessionstorage',
        'indexdb', 'cachestorage', 'serviceworkers',
      ]));
    }

    // Assertion 2 — view destroyed every cycle, never recreated
    expect(destroySpy).toHaveBeenCalledTimes(5);
    expect(createSpy).toHaveBeenCalledTimes(0);

    // Assertion 3 — welcome:show emitted every cycle
    const welcomeShowCalls = sendSpy.mock.calls.filter((c) => c[0] === 'welcome:show');
    expect(welcomeShowCalls.length).toBe(5);

    // Assertion 4 — loop guard NOT tripped
    expect(sessionReset._getStateForTests().loopActive).toBe(false);

    // Assertion 5 — resetTimestamps all welcome mode
    const entries = sessionReset._getStateForTests().resetTimestamps;
    expect(entries.length).toBeLessThanOrEqual(5);
    for (const e of entries) {
      expect(e.mode).toBe('welcome');
      expect(e.reason).toBe('idle-expired');
    }

    // Assertion 6 — listeners fired per cycle
    expect(preSpy).toHaveBeenCalledTimes(5);
    expect(postSpy).toHaveBeenCalledTimes(5);
  });
});
```
