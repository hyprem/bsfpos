```javascript
// src/main/main.js
// Bee Strong POS Kiosk — main process entry.
// Phase 1 plan 02: creates the single kiosk BrowserWindow and wires the
// ipcMain 'cash-register-ready' splash stub. Keyboard lockdown, single-instance
// lock, and globalShortcut registrations are added by plan 03 (REPLACES the
// ORCHESTRATION block below — do NOT move createMainWindow).

const { app, BrowserWindow, Menu, ipcMain } = require('electron');
const path = require('path');
const log = require('./logger');

const isDev = process.env.NODE_ENV === 'development';

// --- createMainWindow -------------------------------------------------------
// Plan 03 imports/consumes this function unchanged. Keep it self-contained.

let mainWindow = null;

function createMainWindow() {
  mainWindow = new BrowserWindow({
    show: false,                      // don't show until ready-to-show → prevents white flash
    kiosk: !isDev,                    // D-07 dev gating
    fullscreen: !isDev,
    frame: isDev,                     // D-07
    autoHideMenuBar: true,
    backgroundColor: '#1A1A1A',       // UI-SPEC dominant color — zero flash at Phase 2 handoff
    paintWhenInitiallyHidden: true,   // ensures ready-to-show fires reliably
    width: isDev ? 420 : undefined,   // D-07 dev dimensions (vertical tablet sim)
    height: isDev ? 800 : undefined,
    resizable: isDev,
    webPreferences: {
      preload: path.join(__dirname, 'preload.js'),
      contextIsolation: true,
      nodeIntegration: false,
      sandbox: true,
      devTools: isDev,                // D-08: prod has no DevTools path
    },
  });

  // Belt + braces vs autoHideMenuBar
  Menu.setApplicationMenu(null);

  mainWindow.once('ready-to-show', () => {
    log.info('mainWindow ready-to-show — showing');
    mainWindow.show();
    if (isDev) {
      mainWindow.webContents.openDevTools({ mode: 'detach' });
    }
  });

  mainWindow.webContents.on('render-process-gone', (_e, details) => {
    log.error('render-process-gone', details);
  });

  mainWindow.loadFile(path.join(__dirname, '..', 'host', 'host.html'));

  // D-03: ipcMain stub for splash lift. Phase 2 fires this from the injection layer.
  // In Phase 1 this handler never fires — splash stays visible forever on a fresh
  // device, which is the correct Phase 1 end state per D-06.
  ipcMain.on('cash-register-ready', () => {
    log.info('cash-register-ready IPC received — lifting splash');
    if (mainWindow && !mainWindow.isDestroyed()) {
      mainWindow.webContents.send('splash:hide');
    }
  });

  return mainWindow;
}

module.exports = { createMainWindow, isDev };

// --- ORCHESTRATION (plan 03 REPLACES everything below this line) -----------

app.whenReady().then(() => {
  log.info('app ready — creating main window (plan 02 orchestration stub)');
  createMainWindow();
});

app.on('window-all-closed', () => {
  app.quit();
});
```

```javascript
// Consumes window.kiosk exposed by preload.js. Phase 1 only handles splash

(function () {
  'use strict';

  if (window.kiosk && window.kiosk.isDev) {

  function hideSplash() {
    const el = document.getElementById('splash');

  function showSplash() {
    const el = document.getElementById('splash');
    if (el) el.style.display = 'flex';

    window.kiosk.onHideSplash(hideSplash);
  }
  if (window.kiosk && window.kiosk.onShowSplash) {
    window.kiosk.onShowSplash(showSplash);
  }
})();
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| host.html renderer → main process | IPC messages cross this boundary; `contextIsolation: true` + `sandbox: true` + `nodeIntegration: false` enforce isolation |
| preload.js → host renderer | Only `contextBridge.exposeInMainWorld('kiosk', ...)` surface is exposed; no ipcRenderer leak |
| Phase 2 BrowserView → host overlay | Future boundary — host splash sits at z-index 100 above any BrowserView mounted at #magicline-mount (z-index 0), so Magicline paint cannot leak visually before splash lifts |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Elevation of Privilege | preload.js contextBridge surface | mitigate | Only expose `isDev`, `onHideSplash`, `onShowSplash`. No `ipcRenderer.send`, no Node APIs, no arbitrary channel access. Audited in acceptance criteria. |
| T-02-02 | Tampering | host.html CSP | mitigate | Explicit `Content-Security-Policy` meta: `default-src 'self'; script-src 'self'; img-src 'self' data:` — blocks remote script injection and external resource loads |
| T-02-03 | Information Disclosure | Splash layer showing stale state | mitigate | `backgroundColor: '#1A1A1A'` + `show: false` + `ready-to-show` eliminate flash-of-white-window before splash paints (SHELL-06 foundation per D-03) |
| T-02-04 | Denial of Service | Splash never lifting | accept | D-06 explicit decision: on a fresh device with no Phase 2/3 wiring, splash stays visible forever as the correct failure mode. No timer fallback. |
| T-02-05 | Spoofing | IPC channel name collision | mitigate | Channel names are namespaced (`splash:hide`, `splash:show`, `cash-register-ready`); future phases must reuse the `kiosk:` / `splash:` prefix. Documented in UI-SPEC. |
</threat_model>
